void create(char n[10])
{
    struct Node *newnode;

    newnode->name=n;
    newnode->addr=NULL;
    if(start==NULL)
    {
        start=newnode;
    }
    else

        curr=start;
        while(curr->addr!=NULL)

            curr=curr->addr;
        }
        curr->addr=newnode;
    }
}

    if(start==NULL)
    {
        printf("linked list is emty");
    }
    else
    {
        curr=start;
        while(curr!=NULL)

            curr=curr->addr;
        }
    }
}
